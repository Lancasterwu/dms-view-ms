\documentclass [11pt, proquest] {uwthesis}[2020/02/24]

% To move figures and tables to the end of the document
\usepackage[nomarkers]{endfloat}
\usepackage{chngcntr}
% To have figures and tables be numbered sequentially over entire document, instead of sequentially by chapter
\counterwithout{figure}{chapter}
\counterwithout{table}{chapter}

\usepackage{natbib}
\usepackage{graphicx}
\usepackage{url}
\usepackage{seqsplit} % to get proper line breaks for long strings such as primer sequences
\usepackage{changepage} % for adjustwidth
\usepackage{setspace} % to set smaller than double-spacing for appendices
\def\bibpreamble{\protect\addcontentsline{toc}{chapter}{Bibliography}}
\usepackage{float} % apparently needs to be loaded so that \begin{figure}[H] actually places the figure "HERE".
\usepackage{amsmath,amssymb} % i think this makes math equations look a little better?
\setcounter{tocdepth}{2}  % Print the chapter and sections to the toc
\usepackage{multirow} % create tabular cells spanning multiple rows

\usepackage[font={small}]{caption} % decrease the size of figure caption fonts
\setlength{\abovecaptionskip}{3pt} % force captions closer to figs, can wiggle with plus 2pt minus 1pt
\setlength{\belowcaptionskip}{3pt}

% Use array to allow text wrapping in tables using the "m{}" argument
\usepackage{array}
\newcolumntype{P}[1]{>{\centering\arraybackslash}p{#1}}

\usepackage{nameref,hyperref}  % linked TOC

% make sure no extra spaces between paragraphs
\setlength{\parskip}{0pt}

% if OK with pro quest, change font:
\renewcommand{\familydefault}{\sfdefault}
\usepackage{helvet}

\newcounter{suppfile} % for the supplemental files

% Puts 'Equation' in front of inline references to equations
% can't do this for tables or figures because then the autobuilt
% `List of Tables`/ `List of Figures` pages
\renewcommand{\theequation}{Equation~\arabic{equation}}

%

     \def\}{\ttfamily\char`\}}}
  {\end{alltt}}
